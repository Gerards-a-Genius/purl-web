## Spacing Scale

Our spacing system uses a consistent scale based on 4px increments:

| Name | Value | CSS Variable | Usage |
|------|-------|--------------|-------|
| **xs** | 4px | `--spacing-xs` | Tight spacing, icon gaps |
| **sm** | 8px | `--spacing-sm` | Compact spacing |
| **md** | 16px | `--spacing-md` | Default padding/margin |
| **lg** | 24px | `--spacing-lg` | Section spacing |
| **xl** | 32px | `--spacing-xl` | Large gaps |
| **2xl** | 48px | `--spacing-2xl` | Page sections |

---

## Border Radius

Rounded corners create a friendly, approachable feel:

| Name | Value | CSS Variable | Usage |
|------|-------|--------------|-------|
| **sm** | 8px | `--radius-sm` | Small components |
| **md** | 12px | `--radius-md` | Default (buttons, inputs) |
| **lg** | 16px | `--radius-lg` | Cards |
| **xl** | 20px | `--radius-xl` | Modals, sheets |
| **2xl** | 24px | `--radius-2xl` | Large containers |
| **3xl** | 32px | `--radius-3xl` | Hero sections |
| **full** | 9999px | `--radius-full` | Pills, avatars |

## Using Tokens

### In Tailwind CSS

```jsx
// Using semantic colors
<button className="bg-primary text-primary-foreground">
  Click me
</button>

// Using brand colors directly
<div className="bg-copper text-white">
  Copper background
</div>

// Using typography classes
<h1 className="text-h1">Page Title</h1>
<p className="text-body text-muted-foreground">Body text</p>
```

### In CSS

```css
.custom-element {
  background: var(--copper);
  color: var(--warm-white);
  border-radius: var(--radius-lg);
  padding: var(--spacing-md);
}
```

export function ColorSwatch({ color, name, cssVar, usage, light }) {
  return (
    <div style={{
      borderRadius: '12px',
      overflow: 'hidden',
      border: '1px solid #E8D5C4',
      background: 'white'
    }}>
      <div style={{
        backgroundColor: color,
        height: '80px',
        display: 'flex',
        alignItems: 'flex-end',
        padding: '8px',
      }}>
        <span style={{
          color: light ? 'white' : '#480903',
          fontSize: '11px',
          fontFamily: 'monospace',
        }}>{color}</span>
      </div>
      <div style={{ padding: '12px' }}>
        <div style={{ fontWeight: 600, fontSize: '14px', color: '#480903' }}>{name}</div>
        <div style={{ fontSize: '12px', color: '#8C6B5D', fontFamily: 'monospace' }}>{cssVar}</div>
        <div style={{ fontSize: '11px', color: '#9C7C78', marginTop: '4px' }}>{usage}</div>
      </div>
    </div>
  );
}

export function TypographySample({ variant, text, details }) {
  const styles = {
    'h1': { fontSize: '28px', fontWeight: 800, letterSpacing: '-0.5px', lineHeight: '34px' },
    'h2': { fontSize: '22px', fontWeight: 700, letterSpacing: '-0.3px', lineHeight: '28px' },
    'h3': { fontSize: '18px', fontWeight: 700, lineHeight: '24px' },
    'body': { fontSize: '16px', fontWeight: 500, lineHeight: '24px' },
    'body-small': { fontSize: '14px', fontWeight: 500, lineHeight: '20px' },
    'label': { fontSize: '12px', fontWeight: 700, letterSpacing: '0.5px', lineHeight: '16px', textTransform: 'uppercase' },
    'caption': { fontSize: '11px', fontWeight: 500, lineHeight: '14px' },
  };
  return (
    <div style={{
      display: 'flex',
      alignItems: 'baseline',
      gap: '16px',
      padding: '12px',
      background: 'white',
      borderRadius: '8px',
      border: '1px solid #E8D5C4'
    }}>
      <span style={{ ...styles[variant], color: '#480903', fontFamily: '"Manrope", system-ui, sans-serif', minWidth: '200px' }}>
        {text}
      </span>
      <span style={{ fontSize: '12px', color: '#8C6B5D', fontFamily: 'monospace' }}>
        {details}
      </span>
    </div>
  );
}
